78 __device__ float dev_slj (const float3 *p1, const float3 *p2, float3 *pairForce, const float3 *box, const float *args, const float *rcut) {

81

83  if (r2 <= args[2]*args[2]) {
84  // shift to just past the singularity and allow to run
