55  }

57  posAtLastBuild_.resize(sys.numAtoms());
58  } catch (std::exception& e) {

60  throw customException ("Unable to initialize position list for neighbor list");
61  return;
62  }
63 
64  for (unsigned int i = 0; i < nlist_index.size(); ++i) {
65  nlist_index[i] = -1;
66  }
67 
68  // must build when initialized
69  build = 1;
70  } else {
71  float drMax1_ = 0.0;
72  float drMax2_ = 0.0;
73  float3 dummy, box = sys.box();
74  for (unsigned int i = 0; i < sys.numAtoms(); ++i) {
75  const float dr2 = pbcDist2 (sys.atoms[i].pos, posAtLastBuild_[i], dummy, box);
76  if (dr2 > drMax1_*drMax1_) {
77  drMax2_ = drMax1_;
78  drMax1_ = sqrt(dr2);
79  } else if (dr2 > drMax2_*drMax2_) {
80  drMax2_ = sqrt(dr2);
81  }
82  }
83  if (drMax1_+drMax2_ > rs_) {
84  build = 1;
85  } else {
86  build = 0;
87  }
88  }
89 
90  // check to rebuild the neighbor list
91  if (build) {
92  std::vector < std::vector < int > > dummyNeighbors (sys.numAtoms());
93  std::vector < int > empty (1, -1);
94  for (unsigned int i = 0; i < dummyNeighbors.size(); ++i) {
95  dummyNeighbors[i] = empty;
96  }

98  std::vector < int > nn(N, 0);
99  const float cut2 = (rs_+rc_)*(rs_+rc_);
100  float3 dummy, box = sys.box();
101  int totalNeighbors = 0;
102  for (unsigned int i = 0; i < N; ++i) {
103  for (unsigned int j = i+i; j < N; ++j) {
104  float dist2 = pbcDist2(sys.atoms[i].pos, sys.atoms[j].pos, dummy, box);
105  if (dist2 < cut2) {
106  nn[i]++;
107  nn[j]++;
108  totalNeighbors += 2;
109 
110  // double the size to reduce the number of times this has to happen
111  if (dummyNeighbors[i].size() < nn[i]) {
112  dummyNeighbors[i].resize(2*dummyNeighbors[i].size());
113  }
114  if (dummyNeighbors[j].size() < nn[j]) {
115  dummyNeighbors[j].resize(2*dummyNeighbors[j].size());
116  }
117  dummyNeighbors[i][nn[i]-1] = j;
118  dummyNeighbors[j][nn[j]-1] = i;
119  }
120  }
121  }

124  try {

126  } catch (std::exception &e) {
127  std::cerr << e.what() << std::endl;

129  }

134  posAtLastBuild_[i] = sys.atoms[i].pos;
135  nlist[counter] = nn[i];
